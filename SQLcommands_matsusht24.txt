CREATE DATABASE CHEESECAKE;
USE CHEESECAKE
CREATE TABLE ORDERS (
	ORDERID int,
	MONTH	char(3),
	DAY int,
	QUANTITY int,
	TOPPING(9) CHECK (TOPPING='plain' OR TOPPING='chocolate' OR TOPPING='cherry'),	
	NOTES varchar(512),
);
DESCRIBE ORDERS;
SELECT * FROM ORDERS;
